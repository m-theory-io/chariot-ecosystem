# Multi-stage Dockerfile for go-chariot
# Optimized for monorepo structure with Azure Key Vault support

# Stage 1: Build stage
FROM golang:1.21-alpine AS builder

# Install build dependencies
RUN apk add --no-cache \
    git \
    ca-certificates \
    tzdata \
    upx

# Set working directory
WORKDIR /src

# Copy go mod files first for better caching
COPY go.mod go.sum ./

# Download dependencies
RUN go mod download && go mod verify

# Copy source code
COPY . .

# Build the application
RUN CGO_ENABLED=0 GOOS=linux GOARCH=amd64 go build \
    -ldflags='-w -s -extldflags "-static"' \
    -a -installsuffix cgo \
    -o bin/go-chariot \
    ./cmd/go-chariot

# Compress binary
RUN upx --lzma bin/go-chariot

# Stage 2: Production stage
FROM alpine:latest AS production

# Install runtime dependencies
RUN apk add --no-cache \
    ca-certificates \
    curl \
    wget \
    netcat-openbsd \
    bash

# Create app user
RUN adduser -D -s /bin/sh appuser

# Set working directory
WORKDIR /app

# Copy binary from builder
COPY --from=builder /src/bin/go-chariot .
COPY --from=builder /src/configs ./configs

# Copy entrypoint and SSL scripts
COPY ../../infrastructure/docker/go-chariot/entrypoint.sh /usr/local/bin/entrypoint.sh
COPY ../../infrastructure/docker/go-chariot/generate-ssl.sh /usr/local/bin/generate-ssl.sh

# Make scripts executable
RUN chmod +x /usr/local/bin/entrypoint.sh /usr/local/bin/generate-ssl.sh

# Create directories
RUN mkdir -p ssl logs config && \
    chown -R appuser:appuser /app

# Switch to app user
USER appuser

# Expose port
EXPOSE 9080

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
  CMD curl -f http://localhost:9080/health || exit 1

# Set entrypoint
ENTRYPOINT ["/usr/local/bin/entrypoint.sh"]
